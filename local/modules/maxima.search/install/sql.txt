CREATE  TABLE IF NOT EXISTS `maxima_search_content`(
  `ID` INT  AUTO_INCREMENT PRIMARY KEY,
  `ITEM_ID` INT,
  `CONTENT_TYPE` VARCHAR(15),
  `TITLE` VARCHAR(255),
  `URL` VARCHAR(255),
  `SEARCH_CONTENT` TEXT,
  `DATE` DATE,
  `DISCIPLINE` VARCHAR(25),
  `FILE_NAME` VARCHAR(100)
 )
ENGINE = MyISAM;
